    Auto   := 0,  // Режим "Автоматический"
    Manual   := 1,  // Режим "Ручной"
    Repair   := 2   // Режим "Ремонт"
);
END_TYPE

	cmdSetModeRepair, // Установить режим "Ремонт"
	cmdSetModeManual, // Установить режим "Ручной"
	cmdSetModeAuto : BOOL; // Установить режим "Авто"

    eStateRemote: E_StateRemote; // Состояние режима управления (авто, ручной, ремонт)

ST_Dumper:
	// DI
	// Обратная связь тормозов
	fbBreakerConveyor, fbBreakerRotation : FB_UniversalSignal; // Тормоз конвейера и поворота (обр связь)
	
	// DO
	// Тормоза
	qxBreakerConveyor, qxBreakerRotation : BOOL; // (1 - Тормоз отпущен, 0 - Тормоз активен)


Требуется осуществить ручной режим управления в ФБ FB_DumperControl

Добавить в ФБ команды переключения в режим. Зависимо от команды, мы переключаем режим автоматический (cmdSetModeAuto) - работа в автоматическом режиме, ручной (cmdSetModeManual) - ручной запуск и остановка через cmdStartConveyor, cmdStopConveyor и ремонт (cmdSetModeRepair) - запрещён какой-либо запуск.

Обновить алгоритм запуска, мы добавляем новый механизм - тормоз мотора для конвейеров и поворота отвалообразователя.

Обновленный алгоритм после подачи сигнала на запуск конвейера: 

    работа ППЗ (при включенной опции) --> Включить сигнал тормоза qxBreakerConveyor (означает отпустить тормоз двигателя) --> Получить обратную связь реле тормоза fbBreakerConveyor.qxSignal --> Включить сигнал ПУСК на VFD от MotorConveyor --> Регулирование частоты с синхронизацией, которое есть сейчас в алгоритме --> Переход в режим WORK
 
 Процесс остановки:
    Обратный порядок, снять сигнал VFD с ЧРП, проверить обратную связь VFD о выключении, снять сигнал с qxBreakerConveyor (поставить на тормоз), проверить обратную связь тормоза - конвейер выключен
